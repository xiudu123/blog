A. Theatre Square

题意

有一个 n × m 的矩形，至少需要多少个大小为 a × a 的正方形能覆盖。

思路

计算覆盖长、宽至少需要的正方形数量，相乘

代码

B. Spreadsheets

题意
两种表示excel表格的方式RXCY（第X行第Y列）、BC23（BC表示列，23表示行）的相互转换。
思路
模拟，26进制和10进制的相互转换
代码
/* When I wrote this, only God and I understood what I was doing Now, only God knows */ #include < bits/stdc++.h > #define int long long typedef long long ll; typedef unsigned long long ull; void slove(); void main_init(); void solve_init(); // ----------------------------- int32_t main(){ std::ios::sync_with_stdio(false); std::cin.tie(nullptr), std::cout.tie(nullptr); std::cout << (std::setiosflags(std::ios::fixed)) << std::setprecision(12); int T = 1; std::cin >> T; main_init(); while(T --){solve_init(), slove();} return 0; } // ---------------------------- void main_init(){} void solve_init(){} void slove(){ std::string str; std::cin >> str; std::function < std::string(void)> work1 = [&]() -> std::string{ int id = str.find('C'); int x = 0; for(int i = id + 1; i < str.size(); ++ i) x = x * 10 + str[i] - '0'; std::string ans; while (x){ int y = ((x - 1) % 26); x = (x - 1) / 26; ans.push_back(char(y + 'A')); } std::reverse(ans.begin(), ans.end()); return ans + str.substr(1, id - 1); }; std::function < std::string(void)> work2 = [&]() -> std::string{ int x = 0, id = 0; for(int i = 0; i < str.size(); ++ i){ id = i; if(std::isdigit(str[i])) break; x = x * 26 + (str[i] - 'A' + 1); } return "R" + str.substr(id) + "C" + std::to_string(x); }; int id = str.find('C', 1); if(str[0] == 'R' && id != std::string::npos && std::isdigit(str[id - 1])){ std::cout << work1() << '\n'; }else std::cout << work2() << '\n'; }
